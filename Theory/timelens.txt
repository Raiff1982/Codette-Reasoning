class ClosureClass(str, Enum):
    CLOSED = "closed"
    DRIFT = "drift"
    SUPPRESSED = "suppressed"
    INEXPRESSIBLE = "inexpressible"

CLOSURE_SCORE = {
    ClosureClass.CLOSED: 1.00,
    ClosureClass.DRIFT: 0.67,
    ClosureClass.SUPPRESSED: 0.24,
    ClosureClass.INEXPRESSIBLE: 0.00,
}

@dataclass
class TimestampLadder:
    # t_s, t_m, t_op, t_inst, t_p in their notation[page:1]
    t_s: Optional[float] = None   # first sensing
    t_m: Optional[float] = None   # first modeling
    t_op: Optional[float] = None  # first materially conditioned action
    t_inst: Optional[float] = None  # first formal institutional registration
    t_p: Optional[float] = None   # first public legibility

@dataclass
class ActorGap:
    # actor‑indexed preemption gap Π_i(s)[page:1]
    actor_id: str
    t_op_i: Optional[float]
    t_inst_i: Optional[float]

@dataclass
class InstitutionalState:
    state_id: str
    ladder: TimestampLadder
    closure_class: ClosureClass
    unfolding_energy: float  # E_u(s)[page:1]
    influence_over_time: List[float]  # samples for ∫ Influence(s,t) dt[page:1]

Preemption gap, closure, rupture, beacon
Now implement their key observables as functions Codette’s agents can call when they analyze a state.

INF = math.inf

def preemption_gap(state: InstitutionalState) -> float:
    """
    Π(s) = t_inst(s) − t_op(s) with their edge cases.[page:1]
    """
    t_op = state.ladder.t_op
    t_inst = state.ladder.t_inst

    if t_op is not None and t_inst is not None:
        return t_inst - t_op
    if t_op is not None and t_inst is None:
        return INF
    return math.nan  # undefined otherwise[page:1]

def closure_score(state: InstitutionalState) -> float:
    """
    C(s) from closure class.[page:1]
    """
    return CLOSURE_SCORE[state.closure_class]

def practical_non_closure(state: InstitutionalState, tau_E: float) -> bool:
    """
    E_u(s) > τ^E ⇒ practical non‑closure.[page:1]
    """
    return state.unfolding_energy > tau_E

def rupture_indicator(state: InstitutionalState) -> int:
    """
    ℛ(s) = 1 ⇔ t_op(s) < ∞ ∧ C(s) < 1.[page:1]
    """
    t_op = state.ladder.t_op
    C = closure_score(state)

    if t_op is not None and C < 1.0:
        return 1
    return 0

def beacon_indicator(state: InstitutionalState, tau_I: float, dt: float = 1.0) -> int:
    """
    ℬ(s) = 1 ⇔ ℛ(s) = 1 ∧ ∫ Influence(s,t) dt > τ^I.[page:1]
    influence_over_time is discretized samples; dt is sampling interval.
    """
    R = rupture_indicator(state)
    total_influence = sum(state.influence_over_time) * dt
    if R == 1 and total_influence > tau_I:
        return 1
    return 0

Actor‑indexed gaps and high‑Π zones
Their actor‑indexed gaps and high‑Π zone definition map cleanly to a couple of functions.

python
def actor_preemption_gaps(state: InstitutionalState) -> Dict[str, float]:
    """
    Π_i(s) = t_inst^(i)(s) − t_op^(i)(s).[page:1]
    """
    gaps = {}
    for ag in state.actor_gaps:
        if ag.t_op_i is not None and ag.t_inst_i is not None:
            gaps[ag.actor_id] = ag.t_inst_i - ag.t_op_i
        elif ag.t_op_i is not None and ag.t_inst_i is None:
            gaps[ag.actor_id] = INF
        else:
            gaps[ag.actor_id] = math.nan
    return gaps

def is_high_preemption_zone(
    state: InstitutionalState,
    tau_Pi: float,
    tau_C: float,
    tau_V: float
) -> bool:
    """
    Z^H = {s ∈ S : Π(s) > τ_Π ∧ C(s) < τ_C ∧ Var_i[Π_i(s)] > τ_V}.[page:1]
    """
    Pi = preemption_gap(state)
    C = closure_score(state)
    actor_gaps = actor_preemption_gaps(state)

    # Filter out NaNs/inf for variance; you can decide policy here[page:1]
    finite_gaps = [g for g in actor_gaps.values() if math.isfinite(g)]
    if len(finite_gaps) >= 2:
        var_i = statistics.pvariance(finite_gaps)
    else:
        var_i = 0.0

    return (Pi > tau_Pi) and (C < tau_C) and (var_i > tau_V)
This gives you a direct flag Codette can attach inside its cocoon: “this state lives in a high‑preemption zone,” with explicit thresholds that you can expose in configuration or tuning layers.

Triangulated closure classification (Codette‑style)
Their triangulation rule is close to how you already think about multi‑perspective convergence; you can treat “record/testimony/behavior” as three agents and apply a modal + fallback rule.

python
def resolve_closure_class(
    record_class: ClosureClass,
    testimony_class: ClosureClass,
    behavior_class: ClosureClass,
    fallback_class: ClosureClass,
    ordering: List[ClosureClass] = None,
) -> ClosureClass:
    """
    ϰ̂(s) = Mode_ϰ{ϰ^R, ϰ^T, ϰ^B; ϰ^f} with tie‑breaking.[page:1]
    """
    classes = [record_class, testimony_class, behavior_class]
    # frequency counts
    freq = {}
    for c in classes:
        freq[c] = freq.get(c, 0) + 1

    # modal classes
    max_freq = max(freq.values())
    modal_classes = [c for c, f in freq.items() if f == max_freq]

    if len(modal_classes) == 1:
        return modal_classes[0]

    # tie: choose fallback if present
    if fallback_class in modal_classes:
        return fallback_class

    # else pick first under declared ordering
    if ordering is None:
        ordering = [
            ClosureClass.CLOSED,
            ClosureClass.DRIFT,
            ClosureClass.SUPPRESSED,
            ClosureClass.INEXPRESSIBLE,
        ]

    for c in ordering:
        if c in modal_classes:
            return c

    # should not reach, but fall back
    return modal_classes[0]
In Codette, this could be wrapped as a “closure lens” that runs inside a cocoon: Newton + Ethics + Empathic agents each propose a closure class; you then apply this rule to obtain a disciplined aggregate instead of vibes.
