#!/usr/bin/env python3
"""
AEGIS Micro-Shield Engine
Kernel-level reachability restriction and syscall filtering for AI Coding Agents.
Created for Jonathan Harrison / Codette Architecture.
"""

import os
import sys
import ctypes
import argparse
import subprocess
from pathlib import Path

# --- Landlock LSM System Call Definitions (Linux Kernel >= 5.13) ---
SYS_LANDLOCK_CREATE_RULESET = 444
SYS_LANDLOCK_ADD_RULE = 445
SYS_LANDLOCK_RESTRICT_SELF = 446

LANDLOCK_ACCESS_FS_EXECUTE = (1 << 0)
LANDLOCK_ACCESS_FS_WRITE_FILE = (1 << 1)
LANDLOCK_ACCESS_FS_READ_FILE = (1 << 2)
LANDLOCK_ACCESS_FS_READ_DIR = (1 << 3)

class LandlockRulesetAttr(ctypes.Structure):
    _fields_ = [
        ("handled_access_fs", ctypes.c_uint64)
    ]

class LandlockPathBeneathAttr(ctypes.Structure):
    _fields_ = [
        ("allowed_access", ctypes.c_uint64),
        ("parent_fd", ctypes.c_int32)
    ]

def restrict_filesystem_landlock(allowed_workspace: Path):
    """Enforces kernel-level filesystem reachability using Linux Landlock LSM."""
    libc = ctypes.CDLL(None)
    
    attr = LandlockRulesetAttr()
    attr.handled_access_fs = (
        LANDLOCK_ACCESS_FS_EXECUTE |
        LANDLOCK_ACCESS_FS_WRITE_FILE |
        LANDLOCK_ACCESS_FS_READ_FILE |
        LANDLOCK_ACCESS_FS_READ_DIR
    )
    
    # Create ruleset
    ruleset_fd = libc.syscall(
        SYS_LANDLOCK_CREATE_RULESET,
        ctypes.byref(attr),
        ctypes.sizeof(attr),
        0
    )
    
    if ruleset_fd < 0:
        print("[!] Warning: Landlock LSM not supported or disabled on this kernel. Falling back to namespace isolation.", file=sys.stderr)
        return False

    # Add rule: Allow Read/Write on designated workspace ONLY
    ws_fd = os.open(allowed_workspace, os.O_PATH | os.O_CLOEXEC)
    path_beneath = LandlockPathBeneathAttr()
    path_beneath.allowed_access = attr.handled_access_fs
    path_beneath.parent_fd = ws_fd

    err = libc.syscall(
        SYS_LANDLOCK_ADD_RULE,
        ruleset_fd,
        1, # LANDLOCK_RULE_PATH_BENEATH
        ctypes.byref(path_beneath),
        0
    )
    os.close(ws_fd)

    if err < 0:
        os.close(ruleset_fd)
        print("[!] Error adding Landlock path rule.", file=sys.stderr)
        return False

    # Enforce ruleset on current process and all future children
    PR_SET_NO_NEW_PRIVS = 38
    libc.prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0)

    res = libc.syscall(SYS_LANDLOCK_RESTRICT_SELF, ruleset_fd, 0)
    os.close(ruleset_fd)

    if res == 0:
        print(f"🔒 [AEGIS Kernel] Landlock restriction active. Boundary locked to: {allowed_workspace}")
        return True
    return False

def sanitize_environment():
    """Scrubs ambient credentials from environment variables."""
    sensitive_keys = [
        "AWS_ACCESS_KEY_ID", "AWS_SECRET_ACCESS_KEY", "AWS_SESSION_TOKEN",
        "AZURE_CLIENT_SECRET", "GITHUB_TOKEN", "GH_TOKEN", "SSH_AUTH_SOCK",
        "GPG_AGENT_INFO", "DATABASE_URL"
    ]
    clean_env = {}
    for key, value in os.environ.items():
        if key not in sensitive_keys and not any(s in key.upper() for s in ["SECRET", "PASS", "PRIVATE_KEY"]):
            clean_env[key] = value
    return clean_env

def run_agent_shielded(target_dir: str, command: list, enable_net: bool = False):
    target_path = Path(target_dir).resolve()

    if target_path == Path.home() or target_path == Path("/"):
        print("[CRITICAL] Safety Tripped: Cannot run agent targeting host Home or Root!", file=sys.stderr)
        sys.exit(1)

    if not target_path.exists():
        print(f"[ERROR] Target path does not exist: {target_path}", file=sys.stderr)
        sys.exit(1)

    clean_env = sanitize_environment()

    # Build unshare command for lightweight kernel namespace isolation
    # -m: Mount namespace
    # -p: PID namespace (prevents seeing host processes)
    # -n: Network namespace (disables network completely if enable_net is False)
    unshare_cmd = ["unshare", "-m", "-p", "-f"]
    if not enable_net:
        unshare_cmd.append("-n")

    print(f"🛡️  [AEGIS Shield] Spawning bounded workspace execution context...")
    
    # Python child initializer to apply Landlock before executing the agent
    def child_preexec():
        restrict_filesystem_landlock(target_path)

    full_cmd = unshare_cmd + (command if command else ["/bin/bash"])

    try:
        subprocess.run(
            full_cmd,
            cwd=target_path,
            env=clean_env,
            preexec_fn=child_preexec if sys.platform.startswith("linux") else None
        )
    except Exception as e:
        print(f"[!] Execution failed or exited: {e}", file=sys.stderr)

def main():
    parser = argparse.ArgumentParser(description="AEGIS Lightweight Agent Defense System")
    parser.add_argument("--dir", "-d", default=".", help="Target project workspace to isolate")
    parser.add_argument("--allow-network", action="store_true", help="Allow agent network access")
    parser.add_argument("cmd", nargs=argparse.REMAINDER, help="Agent command execution array")

    args = parser.parse_args()
    cmd = args.cmd[1:] if args.cmd and args.cmd[0] == "--" else args.cmd

    run_agent_shielded(
        target_dir=args.dir,
        command=cmd,
        enable_net=args.allow_network
    )

if __name__ == "__main__":
    main()
