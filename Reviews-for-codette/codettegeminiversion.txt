import os

class SubstrateAdapter:
    """Represents a single serialized cognitive lens."""
    def __init__(self, name: str, config: dict, base_path: str):
        self.name = name
        self.type = config.get("type", "UNKNOWN")
        self.subfolder = config.get("subfolder", name)
        self.full_path = os.path.join(base_path, self.subfolder)
        self.is_loaded = False
        
    def load_weights(self):
        """Simulates the loading of PEFT weights into the active layer."""
        start_time = time.time()
        if not os.path.exists(self.full_path):
            print(f"[WARNING] Adapter path missing for {self.name}: {self.full_path}")
            return False
            
        # Implementation for peft load_adapter would go here

        load_time = time.time() - start_time
        print(f"    {self.name} [{self.type}] loaded ({load_time:.2f}s)")
        self.is_loaded = True
        return True
